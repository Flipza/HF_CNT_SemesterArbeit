#cloud-config
users:
  - name: ubuntu
    sudo: ALL=(ALL) NOPASSWD:ALL
    groups: users, admin
    home: /home/ubuntu
    shell: /bin/bash
    lock_passwd: false
    plain_text_passwd: 'password'        
# login ssh and console with password
ssh_pwauth: true

packages:
  - unzip
runcmd:
  - sudo snap install microk8s --classic --channel=1.19
  - sudo usermod -a -G microk8s ubuntu
  - sudo microk8s enable dns ingress registry
  - sudo mkdir -p /home/ubuntu/.kube
  - sudo microk8s config >/home/ubuntu/.kube/config
  - sudo chown -f -R ubuntu /home/ubuntu/.kube
  - sudo echo 'alias kubectl="microk8s kubectl"' >>/home/ubuntu/.bashrc 
  - . /etc/os-release
  - sudo sh -c "echo 'deb http://download.opensuse.org/repositories/devel:/kubic:/libcontainers:/stable/x${ID^}_${VERSION_ID}/ /' | sudo -a  /etc/apt/sources.list.d/devel:kubic:libcontainers:stable.list"
  - wget -nv https://download.opensuse.org/repositories/devel:kubic:libcontainers:stable/x${ID^}_${VERSION_ID}/Release.key -O /tmp/Release.key
  - sudo apt-key add - < /tmp/Release.key
  - sudo apt-get update -qq
  - sudo apt-get -qq -y install buildah
  - sudo mkdir /data
  - sudo chmod 777 /data
  - sudo apt-get install -y nfs-common
  - sudo mount -t nfs 10.0.45.8:/data/storage /data
